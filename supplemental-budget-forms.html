[tex]\sf{\small{(angle 1_{\:given\:\: angle}) + (angle 2_{\:its\:\:supplement})=180°}}[/tex]

[tex]\longmapsto \sf{(x+14)+(x)=180°}[/tex]

[tex]\longmapsto \sf{2x+14=180°}[/tex]

[tex]\longmapsto \sf{2x=180-14}[/tex]

[tex]\longmapsto \sf{2x=166}[/tex]

[tex]\longmapsto \boxed{\sf{x=83}}[/tex]

[tex]\rm{\angle{1} = x+14 = 83+14 = 97°}[/tex]
